## How to use ‚öíÔ∏è

```javascript
const { anime, top } = require('animeinfo')

anime('naruto', {
	_type: 'tv',
	_limit: 1
})
	.then(res => console.log(res))


top('manga')
	.then(res => console.log(res))
```
